- [AnyObject method dispatch](https://twitter.com/uint_min/status/1430652663662735362?s=21)

```swift
class Foo {
	let value = 100
  @objc func bar() -> Int {
		return value
	}
}

class Bar {
	let value = 20.0
	@objc func foo() -> Double {
		return value
	}
}

let a: AnyObject = Bar()

if let b = a.bar {
    print("return \(b())")
} else {
    print("Unrecognized selector")
}
```
